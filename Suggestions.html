// تعديل وظائف الاسترجاع، الإضافة، الإعجاب، والحذف لتكون متصلة بالخادم const apiUrl = 'http://localhost:5000/suggestions'; async function renderSuggestions() { const response = await fetch(apiUrl); const suggestions = await response.json(); suggestionContainer.innerHTML = ''; suggestions.forEach((suggestion) => { const suggestionDiv = document.createElement('div'); suggestionDiv.classList.add('suggestion'); suggestionDiv.innerHTML = `

${suggestion.text}

`; suggestionContainer.appendChild(suggestionDiv); }); } async function submitSuggestion() { const text = suggestionInput.value.trim(); if (text) { await fetch(apiUrl, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ text }) }); suggestionInput.value = ''; renderSuggestions(); } else { alert('يرجى كتابة اقتراح.'); } } async function toggleLike(id) { await fetch(`${apiUrl}/${id}/like`, { method: 'PUT' }); renderSuggestions(); } async function deleteSuggestion(id) { const password = prompt('أدخل كلمة السر لحذف الاقتراح:'); if (password === '2007') { await fetch(`${apiUrl}/${id}`, { method: 'DELETE' }); renderSuggestions(); } else { alert('كلمة السر غير صحيحة!'); } }
